#9. Create a script that performs all four basic operations (+, -, *, /) on two provided numbers.

result2=$(echo "$num1 - $num2" | bc) 	
result3=$(echo "$num1 * $num2" | bc)

	echo "Division by 0 not possible"
	exit 1

echo "Division = $result4"

#24. Create a simple calculator where the user enters two numbers and an operator (+ or -) to get the result.

#Simple calculator
#!/bin/bash

read -p "Enter the first number: " num1
read -p "Enter the second number: " num2
read -p "Enter the operator (+,-): " op

#check each condition using if/elif
if [ "$op" == "+" ]; then 	
	result=$(echo "$num1 + $num2" | bc)
elif [ "$op" == "-" ]; then 	
	result=$(echo "$num1 - $num2" | bc)
else 
	echo "Invalid operator selected."
	exit 1
fi

echo "Result = $result"
